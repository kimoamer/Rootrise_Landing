Orchestrator Logic (Python Pseudocode)

def run_orchestrator(session_id):
    session = get_doc("AI Session", session_id)
    agent = session.agent
    
    # Standard Pipeline defined in Config
    pipeline = ["validate_inputs", "diagnostic", "recommendations", "report", "qa"]
    
    for step_name in pipeline:
        instructions = get_agent_instructions(agent, step_name)
        result = execute_step(step_name, session.data, instructions)
        
        session.update_state(step_name, result)
        
        if result.status == "Needs_Input":
            return ask_user(result.question)
            
    session.status = "Completed"
    session.save()
